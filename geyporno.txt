local Players = game:GetService("Players")

local RunService = game:GetService("RunService")

local UserInputService = game:GetService("UserInputService")

local CoreGui = game:GetService("CoreGui")

local TweenService = game:GetService("TweenService")



local LocalPlayer = Players.LocalPlayer

local Camera = workspace.CurrentCamera



local Theme = {

    Bg = Color3.fromRGB(10, 10, 15),

    ElementBg = Color3.fromRGB(20, 20, 25),

    Accent = Color3.fromRGB(0, 255, 255),

    Text = Color3.fromRGB(240, 240, 240),

    TextDim = Color3.fromRGB(160, 160, 160)

}



local State = { Flying = false, HighJump = false, ESP = false, God = true }

local Conf = { FlySpeed = 22, JumpPower = 50, Binds = {} }



-- === UI NOTIFICATION ===

local function ShowNotif(text, color)

    local sg = CoreGui:FindFirstChild("Claudi_UI") or Instance.new("ScreenGui", CoreGui)

    sg.Name = "Claudi_UI"

    local notif = Instance.new("Frame", sg); notif.Size = UDim2.new(0, 200, 0, 40); notif.Position = UDim2.new(0.5, -100, 0, -60)

    notif.BackgroundColor3 = Theme.Bg; Instance.new("UICorner", notif).CornerRadius = UDim.new(0, 8)

    Instance.new("UIStroke", notif).Color = color

    local lbl = Instance.new("TextLabel", notif); lbl.Size = UDim2.new(1,0,1,0); lbl.BackgroundTransparency = 1; lbl.TextColor3 = Theme.Text; lbl.Text = text; lbl.Font = "GothamBold"; lbl.TextSize = 12

    TweenService:Create(notif, TweenInfo.new(0.4, Enum.EasingStyle.Back), {Position = UDim2.new(0.5, -100, 0, 70)}):Play()

    task.delay(1.2, function()

        TweenService:Create(notif, TweenInfo.new(0.4), {Position = UDim2.new(0.5, -100, 0, -60)}):Play()

        task.wait(0.4); notif:Destroy()

    end)

end



-- === MAIN MENU ===

if CoreGui:FindFirstChild("Claudi_Main") then CoreGui.Claudi_Main:Destroy() end

local ScreenGui = Instance.new("ScreenGui", CoreGui); ScreenGui.Name = "Claudi_Main"

local Main = Instance.new("Frame", ScreenGui); Main.Size = UDim2.new(0, 240, 0, 340); Main.Position = UDim2.new(0.5, -120, 0.4, -170)

Main.BackgroundColor3 = Theme.Bg; Main.Visible = false; Instance.new("UICorner", Main).CornerRadius = UDim.new(0, 10)

Instance.new("UIStroke", Main).Color = Theme.Accent



local Scroll = Instance.new("ScrollingFrame", Main); Scroll.Size = UDim2.new(1, -20, 1, -60); Scroll.Position = UDim2.new(0, 10, 0, 50)

Scroll.BackgroundTransparency = 1; Scroll.ScrollBarThickness = 0

Instance.new("UIListLayout", Scroll).Padding = UDim.new(0, 8)



local function CreateBtn(text, stateKey)

    local btn = Instance.new("TextButton", Scroll); btn.Size = UDim2.new(1, 0, 0, 38); btn.BackgroundColor3 = Theme.ElementBg

    btn.Text = text; btn.TextColor3 = Theme.TextDim; btn.Font = "GothamBold"; btn.TextSize = 13; Instance.new("UICorner", btn)

    local bindLbl = Instance.new("TextLabel", btn); bindLbl.Size = UDim2.new(0, 60, 1, 0); bindLbl.Position = UDim2.new(1, -65, 0, 0); bindLbl.BackgroundTransparency = 1; bindLbl.Text = "[---]"; bindLbl.TextColor3 = Theme.Accent; bindLbl.Font = "GothamBold"; bindLbl.TextSize = 10



    btn.MouseButton1Click:Connect(function()

        State[stateKey] = not State[stateKey]

        btn.TextColor3 = State[stateKey] and Theme.Accent or Theme.TextDim

        ShowNotif(text .. (State[stateKey] and " ON" or " OFF"), State[stateKey] and Color3.new(0,1,0) or Color3.new(1,0,0))

    end)



    btn.MouseButton2Click:Connect(function()

        bindLbl.Text = "..."; local c; c = UserInputService.InputBegan:Connect(function(i)

            local key = (i.UserInputType == Enum.UserInputType.MouseButton3 and "MB3") or (i.UserInputType == Enum.UserInputType.Keyboard and tostring(i.KeyCode):gsub("Enum.KeyCode.", ""))

            if key then

                Conf.Binds[stateKey] = i.UserInputType == Enum.UserInputType.MouseButton3 and Enum.UserInputType.MouseButton3 or i.KeyCode

                bindLbl.Text = "[" .. key .. "]"; c:Disconnect()

            end

        end)

    end)

end



local function CreateSlider(text, min, max, def, cb)

    local f = Instance.new("Frame", Scroll); f.Size = UDim2.new(1,0,0,45); f.BackgroundTransparency = 1

    local l = Instance.new("TextLabel", f); l.Size = UDim2.new(1,0,0,15); l.Text = text .. ": " .. def; l.TextColor3 = Theme.Text; l.Font = "GothamBold"; l.TextSize = 11; l.BackgroundTransparency = 1

    local b = Instance.new("Frame", f); b.Size = UDim2.new(1,0,0,4); b.Position = UDim2.new(0,0,0.7,0); b.BackgroundColor3 = Color3.new(0.2,0.2,0.2)

    local fill = Instance.new("Frame", b); fill.Size = UDim2.new((def-min)/(max-min),0,1,0); fill.BackgroundColor3 = Theme.Accent

    b.InputBegan:Connect(function(i)

        if i.UserInputType == Enum.UserInputType.MouseButton1 then

            local c; c = RunService.RenderStepped:Connect(function()

                local p = math.clamp((UserInputService:GetMouseLocation().X - b.AbsolutePosition.X) / b.AbsoluteSize.X, 0, 1)

                fill.Size = UDim2.new(p, 0, 1, 0); local v = math.floor(min + (max-min) * p); l.Text = text .. ": " .. v; cb(v)

            end)

            local endCon; endCon = UserInputService.InputEnded:Connect(function(i2) if i2.UserInputType == Enum.UserInputType.MouseButton1 then c:Disconnect(); endCon:Disconnect() end end)

        end

    end)

end

-- === ЯДРО С ОБНОВЛЕННЫМ УПРАВЛЕНИЕМ ===



RunService.Stepped:Connect(function()

    if State.Flying and LocalPlayer.Character then

        for _, part in pairs(LocalPlayer.Character:GetDescendants()) do

            if part:IsA("BasePart") then part.CanCollide = false end

        end

    end

end)



RunService.Heartbeat:Connect(function(dt)

    local char = LocalPlayer.Character

    local root = char and char:FindFirstChild("HumanoidRootPart")

    local hum = char and char:FindFirstChildOfClass("Humanoid")

    if not root or not hum then return end



    if State.God then

        hum:SetStateEnabled(Enum.HumanoidStateType.Dead, false)

        if hum.Health < 20 then hum.Health = 100 end

    end



    if State.Flying then

        hum.PlatformStand = true

        hum:ChangeState(Enum.HumanoidStateType.Running)



        local moveDir = Vector3.zero

        local look = Camera.CFrame.LookVector

        local right = Camera.CFrame.RightVector



        -- Управление WASD

        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir += Vector3.new(look.X, 0, look.Z) end

        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir -= Vector3.new(look.X, 0, look.Z) end

        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir -= right end

        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir += right end

        

        -- Управление Высотой (Space / LeftShift)

        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir += Vector3.new(0, 1, 0) end

        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir -= Vector3.new(0, 1, 0) end



        if moveDir.Magnitude > 0 then

            root.CFrame = root.CFrame + (moveDir.Unit * Conf.FlySpeed * dt)

        end

        

        root.AssemblyLinearVelocity = Vector3.zero

        root.AssemblyAngularVelocity = Vector3.zero

        root.CFrame = CFrame.new(root.Position, root.Position + Vector3.new(look.X, 0, look.Z))

    else

        if hum.PlatformStand then hum.PlatformStand = false end

    end

end)



-- Чтобы не прыгать при полете

UserInputService.JumpRequest:Connect(function()

    if State.Flying then

        local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")

        if hum then hum:SetStateEnabled(Enum.HumanoidStateType.Jumping, false) end

    elseif State.HighJump then

        local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

        if root then root.AssemblyLinearVelocity = Vector3.new(root.AssemblyLinearVelocity.X, Conf.JumpPower, root.AssemblyLinearVelocity.Z) end

    end

end)



UserInputService.InputBegan:Connect(function(i, g)

    if not g and i.KeyCode == Enum.KeyCode.RightShift then Main.Visible = not Main.Visible end

    if g then return end

    for s, b in pairs(Conf.Binds) do

        if i.KeyCode == b or i.UserInputType == b then

            State[s] = not State[s]

            ShowNotif(s .. (State[s] and " ON" or " OFF"), State[s] and Color3.new(0,1,0) or Color3.new(1,0,0))

        end

    end

end)
